85 void Chronos::Editor::EditorManager::ShapeWindow(){
86  if(this->showShapeWindow){
87  ImGui::Begin("Shape", &this->showShapeWindow);
88  bool doesNameExist = false;
89  ImGui::SeparatorText("Shape Name");
90  ImGui::InputText("Shape Name", this->newShapeParams.shapeName, 200);
91  std::vector<std::pair<int, Chronos::Manager::ShapeParams>> nameShapeDetails = this->manager->getShapeDetails();
92  for(std::pair<int, Chronos::Manager::ShapeParams> shapeDetail : nameShapeDetails){
93  if(!strcmp(this->newShapeParams.shapeName, shapeDetail.second.shapeName)){
94  ImGui::Text("Name Already Exists");
95  doesNameExist = true;
96  break;
97  }
98  }
99  ImGui::SeparatorText("Type");
100  ImGui::RadioButton("Triangle", &this->newShapeType, 0);
101  ImGui::RadioButton("Rectangle", &this->newShapeType, 1);
102  ImGui::SeparatorText("Fill");
103  ImGui::RadioButton("Color", &this->newShapeFill, 0);
104  ImGui::RadioButton("Texture", &this->newShapeFill, 1);
105  ImGui::SeparatorText("Properties");
106  ImGui::DragFloat("X(-1 to 1)", &this->newShapeParams.x, 0.01f, -1.0f, 1.0f);
107  ImGui::DragFloat("Y(-1 to 1)", &this->newShapeParams.y, 0.01f, -1.0f, 1.0f);
108  ImGui::DragFloat("X Size", &this->newShapeParams.xSize, 0.01f, 0.0f, FLT_MAX);
109  ImGui::DragFloat("Y Size", &this->newShapeParams.ySize, 0.01f, 0.0f, FLT_MAX);
110  ImGui::DragFloat("Rotation", &this->newShapeParams.rotation, 0.01f, 0.0f, FLT_MAX);
111  switch(this->newShapeFill){
112  case 0:
113  ImGui::ColorEdit3("Color", this->newShapeParams.color.data());
114  break;
115  case 1:
116  if(ImGui::BeginCombo("Select Texture", currentShapeTextureName)){
117  std::vector<Chronos::Manager::TextureDetails> details = this->manager->getTextureDetails();
118  if(details.size() == 0){
119  ImGui::Selectable("No Textures");
120  } else {
121  for(Chronos::Manager::TextureDetails textureDetail : details){
122  bool isSelected = currentShapeTextureSelection == textureDetail.textureNo;
123  if(ImGui::Selectable(textureDetail.textureName.c_str(), isSelected)){
124  currentShapeTextureSelection = textureDetail.textureNo;
125  strcpy(currentShapeTextureName, textureDetail.textureName.c_str());
126  LOG(1, "Editor", "Texture selected for shapeNo [" + std::to_string(currentShapeTextureSelection) + "], textureName [" + currentShapeTextureName + "]");
127  }
128  if(isSelected){
129  ImGui::SetItemDefaultFocus();
130  }
131  }
132  }
133  ImGui::EndCombo();
134  }
135  }
136  if(!doesNameExist){ \
137  if(ImGui::Button("Add Shape")){
138  Chronos::Manager::PolygonType type;
139  switch(this->newShapeType){
140  case 0:
141  type.triangle = true;
142  break;
143  case 1:
144  type.rectangle = true;
145  break;
146  }
147  switch(this->newShapeFill){
148  case 0:
149  this->manager->addPolygon(this->newShapeParams, type);
150  LOG(1, "Editor", "Colored shape added with shapeName [" + std::string(this->newShapeParams.shapeName) + "]");
151  break;
152  case 1:
153  this->manager->addPolygon(this->newShapeParams, type, this->currentShapeTextureSelection);
154  LOG(1, "Editor", "Textured shape added with shapeName [" + std::string(this->newShapeParams.shapeName) + "]");
155  break;
156  }
157  }
158  }
159  ImGui::SeparatorText("Current Shapes");
160  if(ImGui::BeginListBox("Shapes")){
161  std::vector<std::pair<int, Chronos::Manager::ShapeParams>> shapeDetails = this->manager->getShapeDetails();
162  if(shapeDetails.size() == 0){
163  ImGui::Selectable("No Shapes");
164  } else {
165  for(int i = 0; i < shapeDetails.size(); i++){
166  if(ImGui::Selectable(shapeDetails[i].second.shapeName, this->currentShapeSelection == i)){
167  this->currentShapeSelection = i;
168  }
169  }
170  }
171  ImGui::EndListBox();
172  if(shapeDetails.size() > 0){
173  ImGui::Text("Selected Shape: %d", shapeDetails[currentShapeSelection].first);
174  if(ImGui::Button("Edit Shape")){
175  this->showShapeDetailsWindow = true;
176  this->shapeDetailsShapeNo = shapeDetails[currentShapeSelection].first;
177  this->shapeDetailsShapeParams = shapeDetails[currentShapeSelection].second;
178  }
179  ImGui::SameLine();
180  if(ImGui::Button("Remove Shape")){
181  this->manager->removePolygon(shapeDetails[currentShapeSelection].first);
182  }
183  }
184  }
185 
186  ImGui::End();
187  }

289 void Chronos::Editor::EditorManager::TextureWindow(){
290  if(this->showTextureWindow){
291  ImGui::Begin("Texture", &this->showTextureWindow);
292  bool doesTextureExist = false;
293  ImGui::SeparatorText("Add Textures");
294  ImGui::InputText("Texture Path", this->newTexturePath, 200);
295  ImGui::InputText("Texture Name", this->newTextureName, 200);
296  std::vector<Chronos::Manager::TextureDetails> nameTextureDetails = this->manager->getTextureDetails();
297  for(Chronos::Manager::TextureDetails textureDetail : nameTextureDetails){
298  if(!strcmp(this->newTextureName, textureDetail.textureName.c_str())){
299  ImGui::Text("Name Already Exists");
300  doesTextureExist = true;
301  break;
302  }
303  if(std::string(this->newTextureName).size() == '\0'){
304  ImGui::Text("Name Cannot be empty");
305  doesTextureExist = true;
306  break;
307  }
308  if(!std::string(this->newTexturePath).ends_with(".jpg") && !std::string(this->newTexturePath).ends_with(".png")){
309  ImGui::Text("Only jpg and png files are supported");
310  doesTextureExist = true;
311  break;
312  }
313  }
314 
315  if(!doesTextureExist){
316  if(ImGui::Button("Add Texture")){
317  this->manager->addTexture(this->newTexturePath, this->newTextureName);
318  this->newTexturePath[0] = '\0';
319  LOG(1, "Editor", "Texture added with textureName [" + std::string(this->newTextureName) + "] and texturePath [" + this->newTexturePath + "]");
320  }
321  }
322  ImGui::SeparatorText("Current Textures");
323  std::vector<Chronos::Manager::TextureDetails> details = this->manager->getTextureDetails();
324  if(ImGui::BeginListBox("Textures")){
325  if(details.size() == 0){
326  ImGui::Selectable("No Textures");
327  } else {
328  for(int i = 0; i < details.size(); i++){
329  if(ImGui::Selectable(details[i].textureName.c_str(), currentTextureSelection == i)){
330  currentTextureSelection = i;
331  }
332  }
333  }
334  ImGui::EndListBox();
335  }
336  if(details.size() > 0){
337  if(ImGui::Button("Remove Texture")){
338  this->manager->removeTexture(details[currentTextureSelection].textureNo);
339  LOG(1, "Editor", "Texture removed with textureNo [" + std::to_string(details[currentTextureSelection].textureNo) + "]");
340  }
341  ImGui::SameLine();
342  if(ImGui::Button("View Texture Details")){
343  this->showTextureDetailsWindow = true;
344  LOG(1, "Editor", "Texture selected for editing with textureNo [" + std::to_string(this->textureDetailsCurrentSelection) + "]");
345  }
346  }
347  ImGui::End();
348  }
