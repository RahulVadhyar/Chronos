37 void Chronos::Engine::Polygon::init(Chronos::Engine::Device* device,
38  VkCommandPool commandPool, Chronos::Engine::SwapChain* swapChain,
39  VkSampler textureSampler, Chronos::Engine::Texture texture,
40  VkRenderPass* renderPass, std::vector<std::array<float, 2>> vertices)
41 {
42  this->vertexShaderPath = SPIV_SHADER_PATH "/textureVert.spv";
43  this->fragmentShaderPath = SPIV_SHADER_PATH "/textureFrag.spv";
44  this->texture = texture;
45 
46  // we copy the vertices to the vertex struct. For the texture cooring we use
47  // the same vertex coordinates, but we map them to the range [0, 1] from
48  // [-1, 1]
49  for (auto& vertex : vertices) {
50  TexturedVertex tv = { { vertex[0], vertex[1] },
51  { (vertex[0] + 1.0f) / 2.0f, (vertex[1] + 1.0f) / 2.0f } };
52  this->vertices.push_back(tv);
53  }
54 
55  // we use the earcut library to triangulate the polygon and generate the
56  // indicies
57  std::vector<uint32_t> triangleIndices = mapbox::earcut<uint32_t>(
58  std::array<std::vector<std::array<float, 2>>, 2> { vertices, {} });
59  for (uint32_t i = 0; i < triangleIndices.size(); i++) {
60  if (triangleIndices[i] >= vertices.size()) {
61  throw std::runtime_error(
62  "Index" + std::to_string(triangleIndices[i]) + " out of range");
63  }
64  this->indices.push_back(triangleIndices[i]);
65  }
66 
67  Chronos::Engine::Object::init(
68  device, commandPool, swapChain, textureSampler, renderPass);
69 
70  // create the vertex and index buffers and copy the data
71 
72  polygonVertexBuffers.resize(MAX_FRAMES_IN_FLIGHT);
73  for (auto& buffer : polygonVertexBuffers) {
74  buffer.size = sizeof(this->vertices[0]) * this->vertices.size();
75  buffer.create(*device, VK_BUFFER_USAGE_VERTEX_BUFFER_BIT,
76  VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT
77  | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT); // we use host visible
78  // and coherent memory
79  // to be able to map it
80  // and to edit the
81  // vertiuces
82  copyVerticestoBuffer(buffer);
83  }
84  copyVertices.resize(MAX_FRAMES_IN_FLIGHT, false);
85 
86  indexBuffers.resize(MAX_FRAMES_IN_FLIGHT);
87  for (auto& buffer : indexBuffers) {
88  buffer.size = sizeof(this->indices[0])
89  * (this->vertices.size() - 2); // max indices = n - 2
90  buffer.create(*device, VK_BUFFER_USAGE_INDEX_BUFFER_BIT,
91  VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT
92  | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT); // inidcies will not
93  // change
94  copyIndicestoBuffer(buffer);
95  }
96 }

98 void Chronos::Engine::Polygon::copyVerticestoBuffer(
99  Chronos::Engine::Buffer vertexBuffer)
100 {
101  if (vkMapMemory(device->device, vertexBuffer.memory, 0, VK_WHOLE_SIZE, 0,
102  &this->mappedMemory)
103  != VK_SUCCESS) {
104  throw std::runtime_error("Failed to map memory");
105  }
106  if (this->mappedMemory == nullptr) {
107  throw std::runtime_error("Memory not mapped, current address is "
108  + std::to_string((uint64_t)this->mappedMemory));
109  }
110  memcpy(
111  this->mappedMemory, this->vertices.data(), (size_t)vertexBuffer.size);
112  vkUnmapMemory(device->device, vertexBuffer.memory);
113 }

131 void Chronos::Engine::Polygon::updateVertices(
132  std::vector<std::array<float, 2>> vertices)
133 {
134  // There are a vertex buffer, one for each frame in flight. We need to
135  // update all of them however, some of them may be in use, hence we just set
136  // bools correspoding to each buffer to true on update of the frame, we will
137  // copy the vertices to the buffer if the bool is true for that frame
138  assert(vertices.size() == this->vertices.size());
139  this->vertices.clear();
140  for (auto& vertex : vertices) {
141  TexturedVertex tv = { { vertex[0], vertex[1] },
142  { (vertex[0] + 1.0f) / 2.0f, (vertex[1] + 1.0f) / 2.0f } };
143  this->vertices.push_back(tv);
144  }
145 
146  std::vector<uint32_t> triangleIndices = mapbox::earcut<uint32_t>(
147  std::array<std::vector<std::array<float, 2>>, 2> { vertices, {} });
148  assert(triangleIndices.size() <= this->indices.size());
149  this->indices.clear();
150  this->indices.resize(this->vertices.size() - 2);
151  for (uint32_t i = 0; i < triangleIndices.size(); i++) {
152  if (triangleIndices[i] >= vertices.size()) {
153  throw std::runtime_error(
154  "Index" + std::to_string(triangleIndices[i]) + " out of range");
155  }
156  this->indices.push_back(triangleIndices[i]);
157  }
158 
159  for (int i = 0; i < static_cast<int>(copyVertices.size()); i++) {
160  copyVertices[i] = true;
161  }
162 }

164 void Chronos::Engine::Polygon::update(uint32_t currentFrame)
165 {
166  if (copyVertices[currentFrame]) {
167  copyVerticestoBuffer(polygonVertexBuffers[currentFrame]);
168  copyIndicestoBuffer(indexBuffers[currentFrame]);
169  copyVertices[currentFrame] = false;
170  }
171  uniformBuffers[currentFrame].update(swapChain->swapChainExtent, params.x,
172  params.y, params.rotation, params.xSize, params.ySize);
173 }

186 void Chronos::Engine::Polygon::createDescriptorSets()
187 {
188  std::vector<VkDescriptorSetLayout> layouts(
189  MAX_FRAMES_IN_FLIGHT, descriptorSetLayout);
190  VkDescriptorSetAllocateInfo allocInfo {};
191  allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
192  allocInfo.descriptorPool = descriptorPool;
193  allocInfo.descriptorSetCount = static_cast<uint32_t>(MAX_FRAMES_IN_FLIGHT);
194  allocInfo.pSetLayouts = layouts.data();
195  descriptorSets.resize(MAX_FRAMES_IN_FLIGHT);
196  if (vkAllocateDescriptorSets(
197  device->device, &allocInfo, descriptorSets.data())
198  != VK_SUCCESS) {
199  throw std::runtime_error("failed to allocate descriptor sets!");
200  }
201  for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
202  VkDescriptorBufferInfo bufferInfo {};
203  bufferInfo.buffer = uniformBuffers[i].buffer;
204  bufferInfo.offset = 0;
205  bufferInfo.range = sizeof(UniformBufferObject);
206 
207  VkDescriptorImageInfo imageInfo {};
208  imageInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
209  imageInfo.imageView = texture.textureImageView;
210  imageInfo.sampler = textureSampler;
211 
212  std::array<VkWriteDescriptorSet, 2> descriptorWrites {};
213 
214  descriptorWrites[0].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
215  descriptorWrites[0].dstSet = descriptorSets[i];
216  descriptorWrites[0].dstBinding = 0;
217  descriptorWrites[0].dstArrayElement = 0;
218  descriptorWrites[0].descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
219  descriptorWrites[0].descriptorCount = 1;
220  descriptorWrites[0].pBufferInfo = &bufferInfo;
221 
222  descriptorWrites[1].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
223  descriptorWrites[1].dstSet = descriptorSets[i];
224  descriptorWrites[1].dstBinding = 1;
225  descriptorWrites[1].dstArrayElement = 0;
226  descriptorWrites[1].descriptorType
227  = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
228  descriptorWrites[1].descriptorCount = 1;
229  descriptorWrites[1].pImageInfo = &imageInfo;
230 
231  vkUpdateDescriptorSets(device->device,
232  static_cast<uint32_t>(descriptorWrites.size()),
233  descriptorWrites.data(), 0, nullptr);
234  }
235 }

257  pipelineAttributes.bindingDescriptions.resize(1);

272  pipelineAttributes.colorBlendAttachment.blendEnable = VK_FALSE;
273  pipelineAttributes.colorBlendAttachment.srcColorBlendFactor
