113 void Chronos::Engine::Engine::drawFrame()
114 {
115  // wait for the previous frame to finish
116  glfwPollEvents();
117  vkWaitForFences(device.device, 1, &inFlightFences[currentFrame], VK_TRUE,
118  UINT64_MAX);
119 
120  // get the index of the next image to render to
121  uint32_t imageIndex;
122  VkResult result = vkAcquireNextImageKHR(
123  device.device, swapChain.swapChain, UINT64_MAX,
124  imageAvailableSemaphores[currentFrame], VK_NULL_HANDLE, &imageIndex);
125 
126  // if window has been minimized, then recreate the swap chain and other things
127  if (result == VK_ERROR_OUT_OF_DATE_KHR) {
128  swapChain.recreate();
129  shapeManager.recreate();
130  colorShapeManager.recreate();
131  textManager.recreate();
132 #ifdef ENABLE_EDITOR
133  gui.recreate();
134 #endif
135  return;
136  } else if (result != VK_SUCCESS && result != VK_SUBOPTIMAL_KHR) {
137  throw std::runtime_error("Failed to acquire swap chain image");
138  }
139  // update the shapes and text
140  shapeManager.update(currentFrame);
141  colorShapeManager.update(currentFrame);
142  textManager.update(currentFrame);
143 #ifdef ENABLE_EDITOR
144  gui.update();
145 #endif
146 
147  // reset the fences
148  vkResetFences(device.device, 1, &inFlightFences[currentFrame]);
149 
150  // record the command buffers
151  shapeManager.render(currentFrame, imageIndex, bgColor);
152  colorShapeManager.render(currentFrame, imageIndex, bgColor);
153  textManager.render(currentFrame, imageIndex, bgColor);
154 #ifdef ENABLE_EDITOR
155  gui.render(currentFrame, imageIndex, bgColor);
156 #endif
157 
158  // configure the semaphores
159  VkSemaphore waitSemaphores[] = { imageAvailableSemaphores[currentFrame] };
160  VkPipelineStageFlags waitStages[] = {
161  VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT
162  };
163  VkSemaphore signalSemaphores[] = { renderFinishedSemaphores[currentFrame] };
164 
165  // submit the command buffers
166  std::vector<VkCommandBuffer> submitCommandBuffers;
167  submitCommandBuffers.push_back(shapeManager.commandBuffers[currentFrame]);
168  submitCommandBuffers.push_back(colorShapeManager.commandBuffers[currentFrame]);
169  submitCommandBuffers.push_back(textManager.commandBuffers[currentFrame]);
170 
171 #ifdef ENABLE_EDITOR
172  submitCommandBuffers.push_back(gui.commandBuffers[currentFrame]);
173 #endif
174  VkSubmitInfo submitInfo {};
175  submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
176  submitInfo.waitSemaphoreCount = 1;
177  submitInfo.pWaitSemaphores = waitSemaphores;
178  submitInfo.pWaitDstStageMask = waitStages;
179  submitInfo.commandBufferCount = static_cast<uint32_t>(submitCommandBuffers.size());
180  submitInfo.pCommandBuffers = submitCommandBuffers.data();
181  submitInfo.signalSemaphoreCount = 1;
182  submitInfo.pSignalSemaphores = signalSemaphores;
183 
184  if (vkQueueSubmit(device.graphicsQueue, 1, &submitInfo,
185  inFlightFences[currentFrame])
186  != VK_SUCCESS) {
187  throw std::runtime_error("failed to submit draw command buffer!");
188  }
189 
190 #ifdef ENABLE_EDITOR
191  gui.renderAdditionalViewports();
192 #endif
193  // present the image
194  VkPresentInfoKHR presentInfo {};
195  presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
196  presentInfo.waitSemaphoreCount = 1;
197  presentInfo.pWaitSemaphores = signalSemaphores;
198  VkSwapchainKHR swapChains[] = { swapChain.swapChain };
199  presentInfo.swapchainCount = 1;
200  presentInfo.pSwapchains = swapChains;
201  presentInfo.pImageIndices = &imageIndex;
202  presentInfo.pResults = nullptr;
203  result = vkQueuePresentKHR(device.presentQueue, &presentInfo);
204  if (result == VK_ERROR_OUT_OF_DATE_KHR || result == VK_SUBOPTIMAL_KHR || framebufferResized) {
205  framebufferResized = false;
206  swapChain.recreate();
207  shapeManager.recreate();
208  colorShapeManager.recreate();
209  textManager.recreate();
210 #ifdef ENABLE_EDITOR
211  gui.recreate();
212 #endif
213  } else if (result != VK_SUCCESS) {
214  throw std::runtime_error("Failed to present swap chain image");
215  }
216 
217  // update the current frame
218  currentFrame = (currentFrame + 1) % MAX_FRAMES_IN_FLIGHT;
219 }
