67 int Manager::Manager::addPolygon(Chronos::Manager::ShapeParams shapeParams, Chronos::Manager::PolygonType polygonType, int texture)
68 {
69  //this function is for textured triangles and rectangles only
70  //we need to add a check for the texture
71  std::vector<Chronos::Manager::TextureDetails> textureDetails = getTextureDetails();
72  bool textureExists = false;
73  for (auto& textureDetail : textureDetails) {
74  if (textureDetail.textureNo == texture) {
75  textureExists = true;
76  break;
77  }
78  }
79  if (!textureExists) {
80  throw std::runtime_error("Texture does not exist");
81  }
82 
83  //check whether shape name exists
84  for (auto& shape : engine.shapeManager.objects) {
85  if (strcmp(shape.second.params.shapeName, shapeParams.shapeName) == 0) {
86  throw std::runtime_error("Shape name already exists");
87  }
88  }
89  int shapeNo;
90  if (polygonType.triangle) {
91  shapeNo = engine.shapeManager.addTriangle(shapeParams, engine.textureManager.getTexture(texture));
92  } else if (polygonType.rectangle) {
93  shapeNo = engine.shapeManager.addRectangle(shapeParams, engine.textureManager.getTexture(texture));
94  } else {
95  throw std::runtime_error("Polygon type not supported");
96  }
97 
98  //we add 1 to shapeNo to know they are textured triangles or rectangles
99  shapeNo = std::stoi("1" + std::to_string(shapeNo));
100  return shapeNo;
