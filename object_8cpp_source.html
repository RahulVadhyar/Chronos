36static inline std::vector<char> readFile(const std::string& filename)
37{
38 // reads file. Used to read shaders
39 std::ifstream file(filename, std::ios::ate | std::ios::binary);
40 if (!file.is_open()) {
41 throw std::runtime_error("Failed to open file: " + filename);
42 }
43 size_t fileSize = (size_t)file.tellg();
44 std::vector<char> buffer(fileSize);
45 file.seekg(0);
46 file.read(buffer.data(), fileSize);
47 file.close();
48 return buffer;
49}

62static inline VkShaderModule createShaderModule(
63 const std::vector<char>& code, VkDevice device)
64{
65 VkShaderModuleCreateInfo createInfo {};
66 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;
67 createInfo.codeSize = code.size();
68 createInfo.pCode = reinterpret_cast<const uint32_t*>(code.data());
69 VkShaderModule shaderModule;
70 if (vkCreateShaderModule(device, &createInfo, nullptr, &shaderModule)
71 != VK_SUCCESS) {
72 throw std::runtime_error("Failed to create shader module");
73 }
74 return shaderModule;
75}

99void Chronos::Engine::Object::createGraphicsPipeline()
100{
101 Chronos::Engine::PipelineAttributes pipelineAttributes
102 = getPipelineAttributes();
103
104 auto vertShaderCode = readFile(vertexShaderPath.c_str());
105 auto fragShaderCode = readFile(fragmentShaderPath.c_str());
106
107 VkShaderModule vertShaderModule
108 = createShaderModule(vertShaderCode, device->device);
109 VkShaderModule fragShaderModule
110 = createShaderModule(fragShaderCode, device->device);
111
112 VkPipelineShaderStageCreateInfo vertShaderStageInfo {};
113 vertShaderStageInfo.sType
114 = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
115 vertShaderStageInfo.stage = VK_SHADER_STAGE_VERTEX_BIT;
116 vertShaderStageInfo.module = vertShaderModule;
117 vertShaderStageInfo.pName
118 = "main"; // start from main(vulkan can start from elsewhere also)
119
120 VkPipelineShaderStageCreateInfo fragShaderStageInfo {};
121 fragShaderStageInfo.sType
122 = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
123 fragShaderStageInfo.stage = VK_SHADER_STAGE_FRAGMENT_BIT;
124 fragShaderStageInfo.module = fragShaderModule;
125 fragShaderStageInfo.pName = "main";
126
127 VkPipelineShaderStageCreateInfo shaderStages[]
128 = { vertShaderStageInfo, fragShaderStageInfo };
129
130 auto bindingDescription = pipelineAttributes.bindingDescriptions;
131 auto attributeDescriptions = pipelineAttributes.attributeDescriptions;
132
133 VkPipelineVertexInputStateCreateInfo vertexInputInfo {};
134 vertexInputInfo.sType
135 = VK_STRUCTURE_TYPE_PIPELINE_VERTEX_INPUT_STATE_CREATE_INFO;
136 vertexInputInfo.vertexBindingDescriptionCount
137 = static_cast<uint32_t>(bindingDescription.size());
138 vertexInputInfo.vertexAttributeDescriptionCount
139 = static_cast<uint32_t>(attributeDescriptions.size());
140 vertexInputInfo.pVertexBindingDescriptions = bindingDescription.data();
141 vertexInputInfo.pVertexAttributeDescriptions = attributeDescriptions.data();
142
143 VkPipelineInputAssemblyStateCreateInfo inputAssembly {};
144 inputAssembly.sType
145 = VK_STRUCTURE_TYPE_PIPELINE_INPUT_ASSEMBLY_STATE_CREATE_INFO;
146 inputAssembly.topology
147 = pipelineAttributes.topology; // we are using indexed rendering so
148 // we need to use triangle list
149 inputAssembly.primitiveRestartEnable = VK_FALSE;
150
151 VkViewport viewport {};
152 viewport.x = 0.0f;
153 viewport.y = 0.0f;
154 viewport.width = (float)(*swapChain).swapChainExtent.width;
155 viewport.height = (float)(*swapChain).swapChainExtent.height;
156 viewport.minDepth = 0.0f;
157 viewport.maxDepth = 1.0f;
158
159 VkRect2D scissor {};
160 scissor.offset = { 0, 0 };
161 scissor.extent = (*swapChain).swapChainExtent;
162
163 VkPipelineViewportStateCreateInfo viewportState {};
164 viewportState.sType = VK_STRUCTURE_TYPE_PIPELINE_VIEWPORT_STATE_CREATE_INFO;
165 viewportState.viewportCount = 1;
166 viewportState.pViewports = &viewport;
167 viewportState.scissorCount = 1;
168 viewportState.pScissors = &scissor;
169
170 VkPipelineRasterizationStateCreateInfo rasterizer {};
171 rasterizer.sType
172 = VK_STRUCTURE_TYPE_PIPELINE_RASTERIZATION_STATE_CREATE_INFO;
173 rasterizer.depthClampEnable = VK_FALSE;
174 rasterizer.rasterizerDiscardEnable = VK_FALSE;
175 rasterizer.polygonMode = VK_POLYGON_MODE_FILL;
176 rasterizer.lineWidth = 1.0f;
177 rasterizer.cullMode = VK_CULL_MODE_BACK_BIT;
178 rasterizer.frontFace = pipelineAttributes.frontFace; // counter clockwuise
179 rasterizer.depthBiasEnable = VK_FALSE;
180 rasterizer.depthBiasConstantFactor = 0.0f;
181 rasterizer.depthBiasClamp = 0.0f;
182 rasterizer.depthBiasSlopeFactor = 0.0f;
183
184 VkPipelineMultisampleStateCreateInfo multisampling {};
185 multisampling.sType
186 = VK_STRUCTURE_TYPE_PIPELINE_MULTISAMPLE_STATE_CREATE_INFO;
187 multisampling.sampleShadingEnable = VK_FALSE;
188 multisampling.rasterizationSamples = device->msaaSamples;
189 multisampling.minSampleShading = 1.0f; // Optional
190 multisampling.pSampleMask = nullptr; // Optional
191 multisampling.alphaToCoverageEnable = VK_FALSE; // Optional
192 multisampling.alphaToOneEnable = VK_FALSE;
193 multisampling.sampleShadingEnable = VK_TRUE;
194
195 VkPipelineColorBlendStateCreateInfo colorBlending {};
196 colorBlending.sType
197 = VK_STRUCTURE_TYPE_PIPELINE_COLOR_BLEND_STATE_CREATE_INFO;
198 colorBlending.logicOpEnable = VK_FALSE;
199 colorBlending.logicOp = VK_LOGIC_OP_COPY;
200 colorBlending.attachmentCount = 1;
201 colorBlending.pAttachments = &pipelineAttributes.colorBlendAttachment;
202 colorBlending.blendConstants[0] = 0.0f;
203 colorBlending.blendConstants[1] = 0.0f;
204 colorBlending.blendConstants[2] = 0.0f;
205 colorBlending.blendConstants[3] = 0.0f;
206
207 std::vector<VkDynamicState> dynamicStates
208 = { VK_DYNAMIC_STATE_VIEWPORT, VK_DYNAMIC_STATE_SCISSOR };
209 VkPipelineDynamicStateCreateInfo dynamicState {};
210 dynamicState.sType = VK_STRUCTURE_TYPE_PIPELINE_DYNAMIC_STATE_CREATE_INFO;
211 dynamicState.dynamicStateCount
212 = static_cast<uint32_t>(dynamicStates.size());
213 dynamicState.pDynamicStates = dynamicStates.data();
214
215 VkPipelineLayoutCreateInfo pipelineLayoutInfo {};
216 pipelineLayoutInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_LAYOUT_CREATE_INFO;
217 pipelineLayoutInfo.setLayoutCount = 1; // Optional
218 pipelineLayoutInfo.pSetLayouts = &descriptorSetLayout; // Optional
219 pipelineLayoutInfo.pushConstantRangeCount = 0; // Optional
220 pipelineLayoutInfo.pPushConstantRanges = nullptr; // Optional
221
222 if (vkCreatePipelineLayout(
223 device->device, &pipelineLayoutInfo, nullptr, &pipelineLayout)
224 != VK_SUCCESS) {
225 throw std::runtime_error("failed to create pipeline layout!");
226 }
227
228 VkGraphicsPipelineCreateInfo pipelineInfo {};
229 pipelineInfo.sType = VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO;
230 pipelineInfo.stageCount = 2;
231 pipelineInfo.pStages = shaderStages;
232 pipelineInfo.pVertexInputState = &vertexInputInfo;
233 pipelineInfo.pInputAssemblyState = &inputAssembly;
234 pipelineInfo.pViewportState = &viewportState;
235 pipelineInfo.pRasterizationState = &rasterizer;
236 pipelineInfo.pMultisampleState = &multisampling;
237 pipelineInfo.pDepthStencilState = nullptr; // Optional
238 pipelineInfo.pColorBlendState = &colorBlending;
239 pipelineInfo.pDynamicState = &dynamicState;
240 pipelineInfo.layout = pipelineLayout;
241 pipelineInfo.renderPass = *renderPass;
242 pipelineInfo.subpass = 0;
243 pipelineInfo.basePipelineHandle = VK_NULL_HANDLE; // Optional
244 pipelineInfo.basePipelineIndex = -1;
245
246 if (vkCreateGraphicsPipelines(device->device, VK_NULL_HANDLE, 1,
247 &pipelineInfo, nullptr, &graphicsPipeline)
248 != VK_SUCCESS) {
249 throw std::runtime_error("failed to create graphics pipeline!");
250 }
251
252 // destory the shader modules
253 vkDestroyShaderModule(device->device, fragShaderModule, nullptr);
254 vkDestroyShaderModule(device->device, vertShaderModule, nullptr);
255}

268void Chronos::Engine::Object::createDescriptorPool()
269{
270 std::vector<VkDescriptorType> descriptorTypes = getDescriptorTypes();
271
272 std::vector<VkDescriptorPoolSize> poolSizes { descriptorTypes.size() };
273
274 for (size_t i = 0; i < descriptorTypes.size(); i++) {
275 poolSizes[i].type = descriptorTypes[i];
276 poolSizes[i].descriptorCount
277 = static_cast<uint32_t>(MAX_FRAMES_IN_FLIGHT);
278 }
279 VkDescriptorPoolCreateInfo poolInfo {};
280 poolInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_POOL_CREATE_INFO;
281 poolInfo.poolSizeCount = static_cast<uint32_t>(poolSizes.size());
282 poolInfo.pPoolSizes = poolSizes.data();
283 poolInfo.maxSets = static_cast<uint32_t>(MAX_FRAMES_IN_FLIGHT);
284 if (vkCreateDescriptorPool(
285 device->device, &poolInfo, nullptr, &descriptorPool)
286 != VK_SUCCESS) {
287 throw std::runtime_error("failed to create descriptor pool!");
288 }
289}

291void Chronos::Engine::Object::createDescriptorSetLayout()
292{
293 std::vector<VkDescriptorType> descriptorTypes = getDescriptorTypes();
294 std::vector<VkShaderStageFlagBits> descriptorStages = getDescriptorStages();
295
296 if (descriptorTypes.size() != descriptorStages.size()) {
297 throw std::runtime_error(
298 "descriptorTypes and descriptorStages must be the same size");
299 }
300
301 std::vector<VkDescriptorSetLayoutBinding> bindings {
302 descriptorTypes.size()
303 };
304
305 for (size_t i = 0; i < descriptorTypes.size(); i++) {
306 bindings[i].binding = i;
307 bindings[i].descriptorType = descriptorTypes[i];
308 bindings[i].descriptorCount = 1;
309 bindings[i].stageFlags = descriptorStages[i];
310 bindings[i].pImmutableSamplers = nullptr;
311 }
312
313 VkDescriptorSetLayoutCreateInfo layoutInfo {};
314 layoutInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_LAYOUT_CREATE_INFO;
315 layoutInfo.bindingCount = static_cast<uint32_t>(bindings.size());
316 layoutInfo.pBindings = bindings.data();
317 if (vkCreateDescriptorSetLayout(
318 device->device, &layoutInfo, nullptr, &descriptorSetLayout)
319 != VK_SUCCESS) {
320 throw std::runtime_error("failed to create descriptor set layout!");
321 }
322}
