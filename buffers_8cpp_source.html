45 void Chronos::Engine::Buffer::copy(void* inputData, VkCommandPool commandPool)
46 {
47  // copies the data to a staging buffer in order to copy to device buffer
48  void* data;
49  VkBuffer stagingBuffer;
50  VkDeviceMemory stagingBufferMemory;
51  Chronos::Engine::createBuffer(device, size, VK_BUFFER_USAGE_TRANSFER_SRC_BIT,
52  VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
53  &stagingBuffer, &stagingBufferMemory);
54  vkMapMemory(device.device, stagingBufferMemory, 0, size, 0, &data);
55  memcpy(data, inputData, (size_t)size);
56  vkUnmapMemory(device.device, stagingBufferMemory);
57  Chronos::Engine::copyBuffer(device, stagingBuffer, buffer, size, commandPool);
58  vkDestroyBuffer(device.device, stagingBuffer, nullptr);
59  vkFreeMemory(device.device, stagingBufferMemory, nullptr);
60  LOG(3, "Buffer", "Data [" + std::to_string((uint64_t)buffer) + "] copied to device buffer [" + std::to_string((uint64_t)buffer) + "]");
61 }

78 void Chronos::Engine::UniformBuffer::update(VkExtent2D swapChainExtent, float x, float y,
79  float rotation, float x_size, float y_size)
80 {
81  // This is where the shape parameters are updated
82  UniformBufferObject ubo {};
83  ubo.model = glm::translate(glm::mat4(1.0f), glm::vec3(y, -x, 0.0f));
84  ubo.model = glm::rotate(ubo.model, glm::radians(rotation),
85  glm::vec3(0.0f, 0.0f, 1.0f));
86  ubo.model = glm::scale(ubo.model, glm::vec3(y_size, x_size, 1.0f));
87  ubo.view = glm::lookAt(glm::vec3(0.0f, 0.0f, 0.1f), glm::vec3(0.0f, 0.0f, 0.0f),
88  glm::vec3(1.0f, 0.0f, 0.0f));
89  ubo.proj = glm::ortho(-1.0f, 1.0f,
90  -(float)swapChainExtent.height / (float)swapChainExtent.width,
91  (float)swapChainExtent.height / (float)swapChainExtent.width,
92  -100.0f, 100.0f);
93  ubo.proj[1][1] *= -1;
94  memcpy(data, &ubo, sizeof(ubo));
95  LOG(4, "UniformBuffer", "Uniform buffer object updated for buffer [" + std::to_string((uint64_t)buffer) + "]");
96 }
