25void Chronos::Engine::ColoredRectangle::init(Chronos::Engine::Device* device,
26 VkCommandPool commandPool, Chronos::Engine::SwapChain* swapChain,
27 VkSampler textureSampler, VkRenderPass* renderPass)
28{
29 this->vertexShaderPath = SPIV_SHADER_PATH "/colorVert.spv";
30 this->fragmentShaderPath = SPIV_SHADER_PATH "/colorFrag.spv";
31
32 colorBuffers.resize(MAX_FRAMES_IN_FLIGHT);
33 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
34 colorBuffers[i].create(*device);
35 }
36
37 Chronos::Engine::Object::init(device, commandPool, swapChain,
38 textureSampler, renderPass,
39 Chronos::Engine::ObjectType::TypeColoredRectangle);
40
41 // create the vertex and index buffers and copy the data
42 vertexBuffer.size = sizeof(vertices[0]) * vertices.size();
43 vertexBuffer.create(*device,
44 VK_BUFFER_USAGE_TRANSFER_DST_BIT | VK_BUFFER_USAGE_VERTEX_BUFFER_BIT,
45 VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
46 vertexBuffer.copy((void*)vertices.data(), commandPool);
47
48 indexBuffer.size = sizeof(indices[0]) * indices.size();
49 indexBuffer.create(*device,
50 VK_BUFFER_USAGE_TRANSFER_DST_BIT | VK_BUFFER_USAGE_INDEX_BUFFER_BIT,
51 VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
52 indexBuffer.copy((void*)indices.data(), commandPool);
53}

73void Chronos::Engine::ColoredRectangle::createDescriptorSets()
74{
75 std::vector<VkDescriptorSetLayout> layouts(
76 MAX_FRAMES_IN_FLIGHT, descriptorSetLayout);
77 VkDescriptorSetAllocateInfo allocInfo {};
78 allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
79 allocInfo.descriptorPool = descriptorPool;
80 allocInfo.descriptorSetCount = static_cast<uint32_t>(MAX_FRAMES_IN_FLIGHT);
81 allocInfo.pSetLayouts = layouts.data();
82 descriptorSets.resize(MAX_FRAMES_IN_FLIGHT);
83 if (vkAllocateDescriptorSets(
84 device->device, &allocInfo, descriptorSets.data())
85 != VK_SUCCESS) {
86 throw std::runtime_error("failed to allocate descriptor sets!");
87 }
88 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
89 VkDescriptorBufferInfo bufferInfo {};
90 bufferInfo.buffer = uniformBuffers[i].buffer;
91 bufferInfo.offset = 0;
92 bufferInfo.range = sizeof(UniformBufferObject);
93
94 VkDescriptorBufferInfo bufferInfo2 {};
95 bufferInfo2.buffer = colorBuffers[i].buffer;
96 bufferInfo2.offset = 0;
97 bufferInfo2.range = sizeof(UniformColorBufferObject);
98
99 std::array<VkWriteDescriptorSet, 2> descriptorWrites {};
100
101 descriptorWrites[0].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
102 descriptorWrites[0].dstSet = descriptorSets[i];
103 descriptorWrites[0].dstBinding = 0;
104 descriptorWrites[0].dstArrayElement = 0;
105 descriptorWrites[0].descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
106 descriptorWrites[0].descriptorCount = 1;
107 descriptorWrites[0].pBufferInfo = &bufferInfo;
108
109 descriptorWrites[1].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
110 descriptorWrites[1].dstSet = descriptorSets[i];
111 descriptorWrites[1].dstBinding = 1;
112 descriptorWrites[1].dstArrayElement = 0;
113 descriptorWrites[1].descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
114 descriptorWrites[1].descriptorCount = 1;
115 descriptorWrites[1].pBufferInfo = &bufferInfo2;
116
117 vkUpdateDescriptorSets(device->device,
118 static_cast<uint32_t>(descriptorWrites.size()),
119 descriptorWrites.data(), 0, nullptr);
120 }
121}

137Chronos::Engine::PipelineAttributes
138Chronos::Engine::ColoredRectangle::getPipelineAttributes()
139{
140 Chronos::Engine::PipelineAttributes pipelineAttributes;
141
142 auto bindingDescription = ColorVertex::getBindingDescription();
143 auto attributeDescriptions = ColorVertex::getAttributeDescriptions();
144
145 pipelineAttributes.bindingDescriptions.resize(1);
146 pipelineAttributes.bindingDescriptions[0] = bindingDescription;
147
148 pipelineAttributes.attributeDescriptions.resize(
149 attributeDescriptions.size());
150 for (int i = 0; i < static_cast<int>(attributeDescriptions.size()); i++) {
151 pipelineAttributes.attributeDescriptions[i] = attributeDescriptions[i];
152 }
153
154 pipelineAttributes.topology = VK_PRIMITIVE_TOPOLOGY_TRIANGLE_LIST;
155 pipelineAttributes.frontFace = VK_FRONT_FACE_COUNTER_CLOCKWISE;
156
157 pipelineAttributes.colorBlendAttachment.colorWriteMask
158 = VK_COLOR_COMPONENT_R_BIT | VK_COLOR_COMPONENT_G_BIT
159 | VK_COLOR_COMPONENT_B_BIT | VK_COLOR_COMPONENT_A_BIT;
160 pipelineAttributes.colorBlendAttachment.blendEnable = VK_FALSE;
161 pipelineAttributes.colorBlendAttachment.srcColorBlendFactor
162 = VK_BLEND_FACTOR_ONE;
163 pipelineAttributes.colorBlendAttachment.dstColorBlendFactor
164 = VK_BLEND_FACTOR_ZERO; // Optional
165 pipelineAttributes.colorBlendAttachment.colorBlendOp
166 = VK_BLEND_OP_ADD; // Optional
167 pipelineAttributes.colorBlendAttachment.srcAlphaBlendFactor
168 = VK_BLEND_FACTOR_ONE; // Optional
169 pipelineAttributes.colorBlendAttachment.dstAlphaBlendFactor
170 = VK_BLEND_FACTOR_ZERO; // Optional
171 pipelineAttributes.colorBlendAttachment.alphaBlendOp
172 = VK_BLEND_OP_ADD; // Optional
173
174 return pipelineAttributes;
175}

177void Chronos::Engine::ColoredRectangle::render(uint32_t currentFrame,
178 uint32_t imageIndex, float bgColor[3], VkViewport& viewport,
179 VkRect2D& scissor, std::vector<VkCommandBuffer>& commandBuffers)
180{
181
182 vkCmdBindPipeline(commandBuffers[currentFrame],
183 VK_PIPELINE_BIND_POINT_GRAPHICS, this->graphicsPipeline);
184 vkCmdSetViewport(commandBuffers[currentFrame], 0, 1, &viewport);
185 vkCmdSetScissor(commandBuffers[currentFrame], 0, 1, &scissor);
186 VkBuffer vertexBuffers[] = { this->vertexBuffer.buffer };
187 VkDeviceSize offsets[] = { 0 };
188 vkCmdBindVertexBuffers(
189 commandBuffers[currentFrame], 0, 1, vertexBuffers, offsets);
190 vkCmdBindIndexBuffer(commandBuffers[currentFrame], this->indexBuffer.buffer,
191 0, VK_INDEX_TYPE_UINT16);
192 vkCmdBindDescriptorSets(commandBuffers[currentFrame],
193 VK_PIPELINE_BIND_POINT_GRAPHICS, this->pipelineLayout, 0, 1,
194 &this->descriptorSets[currentFrame], 0, nullptr);
195 vkCmdDrawIndexed(commandBuffers[currentFrame],
196 static_cast<uint32_t>(this->indices.size()), 1, 0, 0, 0);
197}
