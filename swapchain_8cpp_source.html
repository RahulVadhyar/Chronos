29 VkSurfaceFormatKHR Chronos::Engine::chooseSwapSurfaceFormat(
30  const std::vector<VkSurfaceFormatKHR>& availableFormats)
31 {
32  //try to choose SRBG format, else return the first available format
33  for (const auto& availableFormat : availableFormats) {
34  if (availableFormat.format == VK_FORMAT_B8G8R8A8_SRGB && availableFormat.colorSpace == VK_COLOR_SPACE_SRGB_NONLINEAR_KHR) {
35  return availableFormat;
36  }
37  }
38 
39  return availableFormats[0];
40 }

44 VkExtent2D Chronos::Engine::chooseSwapExtent(const VkSurfaceCapabilitiesKHR& capabilities,
45  GLFWwindow* window)
46 {
47  if (capabilities.currentExtent.width != std::numeric_limits<uint32_t>::max()) {
48  return capabilities.currentExtent;
49  } else {
50  int width, height;
51  glfwGetFramebufferSize(window, &width, &height);
52 
53  VkExtent2D actualExtent = { static_cast<uint32_t>(width),
54  static_cast<uint32_t>(height) };
55 
56  actualExtent.width = std::clamp(actualExtent.width, capabilities.minImageExtent.width,
57  capabilities.maxImageExtent.width);
58  actualExtent.height = std::clamp(actualExtent.height, capabilities.minImageExtent.height,
59  capabilities.maxImageExtent.height);
60 
61  return actualExtent;
62  }
63 }

86  details.presentModes.resize(presentModeCount);

103 void Chronos::Engine::SwapChain::create()
104 {
105  // create the swapchain
106  SwapChainSupportDetails swapChainSupport = Chronos::Engine::querySwapChainSupport((*device).physicalDevice, surface);
107  VkSurfaceFormatKHR surfaceFormat = Chronos::Engine::chooseSwapSurfaceFormat(swapChainSupport.formats);
108  VkPresentModeKHR presentMode = this->chooseSwapPresentMode(swapChainSupport.presentModes);
109  VkExtent2D extent = Chronos::Engine::chooseSwapExtent(swapChainSupport.capabilities, window);
110 
111  // set the number of images in the swap chain
112  uint32_t imageCount = swapChainSupport.capabilities.minImageCount + 1;
113 
114  // check if the number of images exceeds the maximum possible images supprted
115  if (swapChainSupport.capabilities.maxImageCount > 0 && imageCount > swapChainSupport.capabilities.maxImageCount) {
116  imageCount = swapChainSupport.capabilities.maxImageCount;
117  }
118 
119  VkSwapchainCreateInfoKHR createInfo {};
120  createInfo.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
121  createInfo.surface = surface;
122  createInfo.minImageCount = imageCount;
123  createInfo.imageFormat = surfaceFormat.format;
124  createInfo.imageColorSpace = surfaceFormat.colorSpace;
125  createInfo.imageExtent = extent;
126  createInfo.imageArrayLayers = 1;
127  createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT;
128 
129  Chronos::Engine::QueueFamilyIndices indices = Chronos::Engine::findQueueFamilies(device->physicalDevice, surface);
130  uint32_t queueFamilyIndices[] = { indices.graphicsFamily.value(),
131  indices.presentFamily.value() };
132 
133  if (indices.graphicsFamily != indices.presentFamily) {
134  createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
135  createInfo.queueFamilyIndexCount = 2;
136  createInfo.pQueueFamilyIndices = queueFamilyIndices;
137  } else {
138  createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
139  createInfo.queueFamilyIndexCount = 0; // Optional
140  createInfo.pQueueFamilyIndices = nullptr; // Optional
141  }
142  createInfo.preTransform = swapChainSupport.capabilities.currentTransform;
143  createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
144  createInfo.presentMode = presentMode;
145  createInfo.clipped = VK_TRUE;
146  createInfo.oldSwapchain = nullptr;
147 
148  if (vkCreateSwapchainKHR(device->device, &createInfo, nullptr, &swapChain) != VK_SUCCESS) {
149  throw std::runtime_error("Failed to create swap chain");
150  }
151  vkGetSwapchainImagesKHR(device->device, swapChain, &imageCount, nullptr);
152  swapChainImages.resize(imageCount);
153  vkGetSwapchainImagesKHR(device->device, swapChain, &imageCount,
154  swapChainImages.data());
155  swapChainImageFormat = surfaceFormat.format;
156  swapChainExtent = extent;
157 }

196  while (width == 0 || height == 0) {

200 
201  vkDeviceWaitIdle(device->device);
202 
203  cleanup();
204  create();
205  createImageViews();

219 VkPresentModeKHR Chronos::Engine::SwapChain::chooseSwapPresentMode(
220  const std::vector<VkPresentModeKHR>& availablePresentModes)
221 {
222  // The presentation modes available
223  // VK_PRESENT_MODE_IMMEDIATE_KHR: Images submitted by your application are
224  // transferred to the screen right away, which may result in tearing.
225  // VK_PRESENT_MODE_FIFO_KHR : The swap chain is a queue where the display
226  // takes an image from the front of the queue when the display is refreshed
227  // and the program inserts rendered images at the
228  // back of the queue.If the queue is full then the program has to wait. This is
229  // most similar to vertical sync as found in modern games.The moment that the
230  // display is refreshed is known as "vertical blank".
231  // VK_PRESENT_MODE_FIFO_RELAXED_KHR : This mode only differs from the previous
232  // one if the application is late and the queue was empty at the last vertical
233  // blank.Instead of waiting for the next vertical blank, the image is
234  // transferred right away when it finally arrives.This may result in visible
235  // tearing. VK_PRESENT_MODE_MAILBOX_KHR : This is another variation of the
236  // second mode.Instead of blocking the application when the queue is full, the
237  // images that are already queued are simply replaced with the newer ones.This
238  // mode can be used to render frames as fast as possible while still avoiding
239  // tearing, resulting in fewer latency issues than standard vertical sync.This
240  // is commonly known as "triple buffering", although the existence of three
241  // buffers alone does not necessarily mean that the framerate is unlocked.
242 
243  //try to select mailbox mode, else select the default FIFO mode
244  for (const auto& availablePresentMode : availablePresentModes) {
245  if (availablePresentMode == this->preferredPresentMode) {
246  return availablePresentMode;
247  }
248  }
249 
250  return VK_PRESENT_MODE_FIFO_KHR;
251 }
