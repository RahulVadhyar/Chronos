48static inline bool checkDeviceExtensionSupport(VkPhysicalDevice device)
49{
50 uint32_t extensionCount;
51 vkEnumerateDeviceExtensionProperties(
52 device, nullptr, &extensionCount, nullptr);
53
54 std::vector<VkExtensionProperties> availableExtensions(extensionCount);
55 vkEnumerateDeviceExtensionProperties(
56 device, nullptr, &extensionCount, availableExtensions.data());
57
58 std::set<std::string> requiredExtensions(
59 Chronos::Engine::deviceExtensions.begin(), Chronos::Engine::deviceExtensions.end());
60
61 for (const auto& extension : availableExtensions) {
62 requiredExtensions.erase(extension.extensionName);
63 }
64
65 return requiredExtensions.empty();
66}

81static inline bool isDeviceSuitable(
82 VkPhysicalDevice device, VkSurfaceKHR surface)
83{
84 // check if the device has the required queue families
85 Chronos::Engine::QueueFamilyIndices indices
86 = Chronos::Engine::findQueueFamilies(device, surface);
87 bool extensionsSupported
88 = checkDeviceExtensionSupport(device);
89
90 bool swapChainAdequate = false;
91 if (extensionsSupported) {
92 Chronos::Engine::SwapChainSupportDetails swapChainSupport
93 = Chronos::Engine::querySwapChainSupport(device, surface);
94 swapChainAdequate = !swapChainSupport.formats.empty()
95 && !swapChainSupport.presentModes.empty();
96 }
97
98 VkPhysicalDeviceFeatures supportedFeatures;
99 vkGetPhysicalDeviceFeatures(device, &supportedFeatures);
100 return indices.isComplete() && extensionsSupported && swapChainAdequate
101 && supportedFeatures.samplerAnisotropy;
102}

114 VkSampleCountFlagBits getMaxUsableSampleCount(
115 VkPhysicalDevice physicalDevice)
116{
117 // used to get max MSAA count
118 VkPhysicalDeviceProperties physicalDeviceProperties;
119 vkGetPhysicalDeviceProperties(physicalDevice, &physicalDeviceProperties);
120
121 VkSampleCountFlags counts
122 = physicalDeviceProperties.limits.framebufferColorSampleCounts;
123 if (counts & VK_SAMPLE_COUNT_64_BIT) {
124 return VK_SAMPLE_COUNT_64_BIT;
125 }
126 if (counts & VK_SAMPLE_COUNT_32_BIT) {
127 return VK_SAMPLE_COUNT_32_BIT;
128 }
129 if (counts & VK_SAMPLE_COUNT_16_BIT) {
130 return VK_SAMPLE_COUNT_16_BIT;
131 }
132 if (counts & VK_SAMPLE_COUNT_8_BIT) {
133 return VK_SAMPLE_COUNT_8_BIT;
134 }
135 if (counts & VK_SAMPLE_COUNT_4_BIT) {
136 return VK_SAMPLE_COUNT_4_BIT;
137 }
138 if (counts & VK_SAMPLE_COUNT_2_BIT) {
139 return VK_SAMPLE_COUNT_2_BIT;
140 }
141 return VK_SAMPLE_COUNT_1_BIT;
142}

144void Chronos::Engine::Device::pickPhysicalDevice(
145 VkInstance instance, VkSurfaceKHR surface)
146{
147 // get the number of devices
148 uint32_t deviceCount = 0;
149 vkEnumeratePhysicalDevices(instance, &deviceCount, nullptr);
150 if (deviceCount == 0) {
151 LOG(1, "Device", "Failed to find GPUs with Vulkan support")
152 throw std::runtime_error("Failed to find GPUs with Vulkan support");
153 }
154 LOG(3, "Device", "Number of devices found: " + std::to_string(deviceCount))
155 std::vector<VkPhysicalDevice> devices(deviceCount);
156 vkEnumeratePhysicalDevices(instance, &deviceCount, devices.data());
157
158 // check if the device is suitable
159 for (const auto& device : devices) {
160 if (isDeviceSuitable(device, surface)) {
161 physicalDevice = device;
162 maxMsaaSamples = getMaxUsableSampleCount(device);
163 break;
164 }
165 }
166 if (physicalDevice == VK_NULL_HANDLE) {
167 LOG(1, "Device", "Failed to find a suitable GPU")
168 throw std::runtime_error("Failed to find a suitable GPU");
169 }
170#ifdef CHRONOS_ENABLE_LOGGING
171 VkPhysicalDeviceProperties deviceProperties;
172 vkGetPhysicalDeviceProperties(physicalDevice, &deviceProperties);
173 LOG(3, "Device", "****PHYSICAL DEVICE INFORMATION*****")
174 LOG(3, "Device",
175 "Physical device picked: " + std::string(deviceProperties.deviceName))
176 LOG(3, "Device",
177 "Driver version: " + std::to_string(deviceProperties.driverVersion))
178 LOG(3, "Device",
179 "API version: " + std::to_string(deviceProperties.apiVersion))
180 LOG(3, "Device", "Device ID: " + std::to_string(deviceProperties.deviceID))
181 LOG(3, "Device", "Vendor ID: " + std::to_string(deviceProperties.vendorID))
182 LOG(3, "Device", "MSAA samples: " + std::to_string(msaaSamples))
183#endif
184}

186void Chronos::Engine::Device::createLogicalDevice(VkSurfaceKHR surface)
187{
188 QueueFamilyIndices indices = findQueueFamilies(physicalDevice, surface);
189
190 std::vector<VkDeviceQueueCreateInfo> queueCreateInfos;
191 std::set<uint32_t> uniqueQueueFamilies
192 = { indices.graphicsFamily.value(), indices.presentFamily.value() };
193
194 float queuePriority = 1.0f;
195 for (uint32_t queueFamily : uniqueQueueFamilies) {
196 VkDeviceQueueCreateInfo queueCreateInfo {};
197 queueCreateInfo.sType = VK_STRUCTURE_TYPE_DEVICE_QUEUE_CREATE_INFO;
198 queueCreateInfo.queueFamilyIndex = queueFamily;
199 queueCreateInfo.queueCount = 1;
200 queueCreateInfo.pQueuePriorities = &queuePriority;
201 queueCreateInfos.push_back(queueCreateInfo);
202 }
203
204 VkPhysicalDeviceFeatures deviceFeatures {};
205 deviceFeatures.samplerAnisotropy = VK_TRUE;
206 deviceFeatures.sampleRateShading = VK_TRUE;
207
208 VkDeviceCreateInfo createInfo {};
209 createInfo.sType = VK_STRUCTURE_TYPE_DEVICE_CREATE_INFO;
210 createInfo.pQueueCreateInfos = queueCreateInfos.data();
211 createInfo.queueCreateInfoCount
212 = static_cast<uint32_t>(queueCreateInfos.size());
213 createInfo.pEnabledFeatures = &deviceFeatures;
214 createInfo.enabledExtensionCount
215 = static_cast<uint32_t>(deviceExtensions.size());
216 createInfo.ppEnabledExtensionNames = deviceExtensions.data();
217 createInfo.enabledLayerCount = 0;
218#ifdef ENABLE_VALIDATION_LAYERS
219 createInfo.enabledLayerCount
220 = static_cast<uint32_t>(validationLayers.size());
221 createInfo.ppEnabledLayerNames = validationLayers.data();
222#endif
223
224 if (vkCreateDevice(physicalDevice, &createInfo, nullptr, &device)
225 != VK_SUCCESS) {
226 LOG(1, "Device", "Failed to create logical device")
227 throw std::runtime_error("Failed to create logical device");
228 }
229
230 LOG(3, "Device", "Logical device created")
231 // get the queue handle
232 vkGetDeviceQueue(device, indices.graphicsFamily.value(), 0, &graphicsQueue);
233 vkGetDeviceQueue(device, indices.presentFamily.value(), 0, &presentQueue);
234}
