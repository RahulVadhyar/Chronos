27void Chronos::Engine::Text::init(Chronos::Engine::Device* device,
28 VkCommandPool commandPool, Chronos::Engine::SwapChain* swapChain,
29 VkSampler textureSampler, VkRenderPass* renderPass,
30 Chronos::Engine::FontTypes fontStyle)
31{
32 vertexShaderPath = SPIV_SHADER_PATH "/textVert.spv";
33 fragmentShaderPath = SPIV_SHADER_PATH "/textFrag.spv";
34
35 // set the fontstyle
36 this->fontStyle = fontStyle;
37
38 // initalize the font
39 uint32_t fontWidth = fontStyle.fontWidth;
40 uint32_t fontHeight = fontStyle.fontHeight;
41 firstChar = fontStyle.firstChar;
42
43 // unsigned char fontpixels[fontHeight][256];
44 //this causes a memory leak... too bad, MSVC is fucked
45 //MSVC can't take variable multi dimensional arrays, else the above commented line would have worked.
46 fontpixels = new unsigned char[fontHeight][256];
47 fontStyle.getFontData(stbFontData, fontpixels, fontHeight);
48
49 // create the font texture from the raw data
50 fontTexture.create(*device, commandPool, (void*)&(fontpixels)[0][0],
51 static_cast<size_t>(fontWidth), static_cast<size_t>(fontHeight),
52 static_cast<VkDeviceSize>(fontWidth * fontHeight), VK_FORMAT_R8_UNORM,
53 "Text Texture");
54
55 // create the vertex buffer
56 VkDeviceSize bufferSize = maxTextLength * sizeof(glm::vec4);
57 Chronos::Engine::createBuffer(*device, bufferSize,
58 VK_BUFFER_USAGE_VERTEX_BUFFER_BIT,
59 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT
60 | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT, // want to be visible to
61 // host and device
62 &vertexBuffer, &vertexBufferMemory);
63
64 // create the uniform buffers for color changing
65 colorBuffers.resize(MAX_FRAMES_IN_FLIGHT);
66 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
67 colorBuffers[i].create(*device);
68 }
69 Chronos::Engine::Object::init(
70 device, commandPool, swapChain, textureSampler, renderPass, Chronos::Engine::ObjectType::TypeText);
71}

87void Chronos::Engine::Text::createDescriptorSets()
88{
89 std::vector<VkDescriptorSetLayout> layouts(
90 MAX_FRAMES_IN_FLIGHT, descriptorSetLayout);
91 VkDescriptorSetAllocateInfo allocInfo {};
92 allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
93 allocInfo.descriptorPool = descriptorPool;
94 allocInfo.descriptorSetCount = static_cast<uint32_t>(MAX_FRAMES_IN_FLIGHT);
95 allocInfo.pSetLayouts = layouts.data();
96 descriptorSets.resize(MAX_FRAMES_IN_FLIGHT);
97 if (vkAllocateDescriptorSets(
98 device->device, &allocInfo, descriptorSets.data())
99 != VK_SUCCESS) {
100 throw std::runtime_error("failed to allocate descriptor sets!");
101 }
102 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
103
104 VkDescriptorImageInfo imageInfo {};
105 imageInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
106 imageInfo.imageView = fontTexture.textureImageView;
107 imageInfo.sampler = textureSampler;
108
109 VkDescriptorBufferInfo bufferInfo {};
110 bufferInfo.buffer = uniformBuffers[i].buffer;
111 bufferInfo.offset = 0;
112 bufferInfo.range = sizeof(UniformBufferObject);
113
114 VkDescriptorBufferInfo bufferInfo2 {};
115 bufferInfo2.buffer = colorBuffers[i].buffer;
116 bufferInfo2.offset = 0;
117 bufferInfo2.range = sizeof(UniformColorBufferObject);
118
119 std::array<VkWriteDescriptorSet, 3> descriptorWrites {};
120
121 descriptorWrites[0].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
122 descriptorWrites[0].dstSet = descriptorSets[i];
123 descriptorWrites[0].dstBinding = 0;
124 descriptorWrites[0].dstArrayElement = 0;
125 descriptorWrites[0].descriptorType
126 = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
127 descriptorWrites[0].descriptorCount = 1;
128 descriptorWrites[0].pImageInfo = &imageInfo;
129
130 descriptorWrites[1].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
131 descriptorWrites[1].dstSet = descriptorSets[i];
132 descriptorWrites[1].dstBinding = 1;
133 descriptorWrites[1].dstArrayElement = 0;
134 descriptorWrites[1].descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
135 descriptorWrites[1].descriptorCount = 1;
136 descriptorWrites[1].pBufferInfo = &bufferInfo;
137
138 descriptorWrites[2].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
139 descriptorWrites[2].dstSet = descriptorSets[i];
140 descriptorWrites[2].dstBinding = 2;
141 descriptorWrites[2].dstArrayElement = 0;
142 descriptorWrites[2].descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
143 descriptorWrites[2].descriptorCount = 1;
144 descriptorWrites[2].pBufferInfo = &bufferInfo2;
145
146 vkUpdateDescriptorSets(device->device,
147 static_cast<uint32_t>(descriptorWrites.size()),
148 descriptorWrites.data(), 0, nullptr);
149 }
150};

215void Chronos::Engine::Text::updateBuffer()
216{
217 // update the vertex buffer with new text on update
218 if (vkMapMemory(device->device, vertexBufferMemory, 0, VK_WHOLE_SIZE, 0,
219 (void**)&mappedMemory)
220 != VK_SUCCESS) {
221 throw std::runtime_error("failed to map memory!");
222 }
223
224 numLetters = 0;
225
226 assert(mappedMemory != nullptr);
227
228 const float charW = 1.5f * params.scale / swapChain->swapChainExtent.width;
229 const float charH = 1.5f * params.scale / swapChain->swapChainExtent.height;
230
231 float fbW = (float)swapChain->swapChainExtent.width;
232 float fbH = (float)swapChain->swapChainExtent.height;
233 float x = (0 / fbW * 2.0f);
234 float y = (0 / fbH * 2.0f);
235
236 // Calculate text width
237 float textWidth = 0;
238 for (auto letter : params.text) {
239 stb_fontchar* charData = &stbFontData[(uint32_t)letter - firstChar];
240 textWidth += charData->advance * charW;
241 }
242 // switch (alignment) {
243 // case Right:
244 // x -= textWidth;
245 // break;
246 // case Center:
247 x -= textWidth / 2.0f;
248 // break;
249 // case Left:
250 // break;
251 // }
252
253 // Generate a uv mapped quad per char in the new text
254 for (auto letter : params.text) {
255 stb_fontchar* charData = &stbFontData[(uint32_t)letter - firstChar];
256
257 mappedMemory->x = (x + (float)charData->x0 * charW);
258 mappedMemory->y = (y + (float)charData->y0 * charH);
259 mappedMemory->z = charData->s0;
260 mappedMemory->w = charData->t0;
261 mappedMemory++;
262
263 mappedMemory->x = (x + (float)charData->x1 * charW);
264 mappedMemory->y = (y + (float)charData->y0 * charH);
265 mappedMemory->z = charData->s1;
266 mappedMemory->w = charData->t0;
267 mappedMemory++;
268
269 mappedMemory->x = (x + (float)charData->x0 * charW);
270 mappedMemory->y = (y + (float)charData->y1 * charH);
271 mappedMemory->z = charData->s0;
272 mappedMemory->w = charData->t1;
273 mappedMemory++;
274
275 mappedMemory->x = (x + (float)charData->x1 * charW);
276 mappedMemory->y = (y + (float)charData->y1 * charH);
277 mappedMemory->z = charData->s1;
278 mappedMemory->w = charData->t1;
279 mappedMemory++;
280
281 x += charData->advance * charW;
282 numLetters++;
283 }
284
285 vkUnmapMemory(device->device, vertexBufferMemory);
286 mappedMemory = nullptr;
287}

312void Chronos::Engine::Text::render(uint32_t currentFrame, uint32_t imageIndex, float bgColor[3], VkViewport& viewport, VkRect2D& scissor, std::vector<VkCommandBuffer>& commandBuffers){
313 vkCmdBindPipeline(commandBuffers[currentFrame],
314 VK_PIPELINE_BIND_POINT_GRAPHICS, this->graphicsPipeline);
315 vkCmdSetViewport(commandBuffers[currentFrame], 0, 1, &viewport);
316 vkCmdSetScissor(commandBuffers[currentFrame], 0, 1, &scissor);
317 VkBuffer vertexBuffers[] = { this->vertexBuffer };
318 VkDeviceSize offsets[] = { 0 };
319 vkCmdBindVertexBuffers(
320 commandBuffers[currentFrame], 0, 1, vertexBuffers, offsets);
321 vkCmdBindVertexBuffers(
322 commandBuffers[currentFrame], 1, 1, vertexBuffers, offsets);
323 vkCmdBindDescriptorSets(commandBuffers[currentFrame],
324 VK_PIPELINE_BIND_POINT_GRAPHICS, this->pipelineLayout, 0, 1,
325 &this->descriptorSets[currentFrame], 0, nullptr);
326 for (uint32_t j = 0; j < this->numLetters; j++) {
327 vkCmdDraw(commandBuffers[currentFrame], 4, 1, j * 4, 0);
328 }
