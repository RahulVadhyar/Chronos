39 void Chronos::Engine::createImage(Chronos::Engine::Device device, uint32_t width, uint32_t height,
40  VkFormat format, VkImageTiling tiling, VkImageUsageFlags usage,
41  VkMemoryPropertyFlags properties, VkImage* image,
42  VkDeviceMemory* imageMemory,
43  VkSampleCountFlagBits numSamples)
44 {
45  VkImageCreateInfo imageInfo {};
46  imageInfo.sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO;
47  imageInfo.imageType = VK_IMAGE_TYPE_2D;
48  imageInfo.extent.width = width;
49  imageInfo.extent.height = height;
50  imageInfo.extent.depth = 1;
51  imageInfo.mipLevels = 1;
52  imageInfo.arrayLayers = 1;
53  imageInfo.format = format;
54  imageInfo.tiling = tiling;
55  imageInfo.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
56  imageInfo.usage = usage;
57  imageInfo.samples = numSamples;
58  imageInfo.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
59 
60  if (vkCreateImage(device.device, &imageInfo, nullptr, image) != VK_SUCCESS) {
61  throw std::runtime_error("failed to create image!");
62  }
63 
64  VkMemoryRequirements memRequirements;
65  vkGetImageMemoryRequirements(device.device, *image, &memRequirements);
66 
67  VkMemoryAllocateInfo allocInfo {};
68  allocInfo.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO;
69  allocInfo.allocationSize = memRequirements.size;
70  allocInfo.memoryTypeIndex = Chronos::Engine::findMemoryType(memRequirements.memoryTypeBits,
71  properties, device.physicalDevice);
72 
73  if (vkAllocateMemory(device.device, &allocInfo, nullptr, imageMemory) != VK_SUCCESS) {
74  throw std::runtime_error("failed to allocate image memory!");
75  }
76 
77  vkBindImageMemory(device.device, *image, *imageMemory, 0);
78 }
79 void Chronos::Engine::Texture::create(Chronos::Engine::Device device, VkCommandPool commandPool,
80  std::string texturePath, std::string textureName)
81 {
82  this->texturePath = texturePath;
83  this->textureName = textureName;
84  int texWidth = 0, texHeight = 0, texChannels = 0;
85  this->device = device;
86  VkDeviceSize imageSize;
87  uint8_t* pixels;
88 
89  if (texturePath.ends_with(".png") || texturePath.ends_with(".jpg")) {
90  pixels = stbi_load(texturePath.c_str(), &texWidth, &texHeight, &texChannels,
91  STBI_rgb_alpha);
92  imageSize = texWidth * texHeight * 4;
93  this->width = texWidth;
94  this->height = texHeight;
95  if (!pixels) {
96  throw std::runtime_error("failed to load texture image " + texturePath);
97  }
98  } else {
99  throw std::runtime_error("unsupported texture format " + texturePath);
100  }
101 
102  VkBuffer stagingBuffer;
103  VkDeviceMemory stagingBufferMemory;
104  Chronos::Engine::createBuffer(device, imageSize, VK_BUFFER_USAGE_TRANSFER_SRC_BIT,
105  VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
106  &stagingBuffer, &stagingBufferMemory);
107 
108  void* data;
109  vkMapMemory(device.device, stagingBufferMemory, 0, imageSize, 0, &data);
110  memcpy(data, pixels, static_cast<size_t>(imageSize));
111  vkUnmapMemory(device.device, stagingBufferMemory);
112 
113  stbi_image_free(pixels);
114 
115  Chronos::Engine::createImage(device, texWidth, texHeight, VK_FORMAT_R8G8B8A8_SRGB,
116  VK_IMAGE_TILING_OPTIMAL,
117  VK_IMAGE_USAGE_TRANSFER_DST_BIT | VK_IMAGE_USAGE_SAMPLED_BIT,
118  VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT, &textureImage,
119  &textureImageMemory, VK_SAMPLE_COUNT_1_BIT);
120 
121  Chronos::Engine::transitionImageLayout(
122  textureImage, VK_IMAGE_LAYOUT_UNDEFINED,
123  VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, commandPool, device);
124  Chronos::Engine::copyBufferToImage(stagingBuffer, textureImage,
125  static_cast<uint32_t>(texWidth),
126  static_cast<uint32_t>(texHeight), commandPool, device);
127  Chronos::Engine::transitionImageLayout(textureImage,
128  VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
129  VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL, commandPool,
130  device);
131  vkDestroyBuffer(device.device, stagingBuffer, nullptr);
132  vkFreeMemory(device.device, stagingBufferMemory, nullptr);
133  textureImageView = createImageView(device, VK_FORMAT_R8G8B8A8_SRGB, textureImage);
134  #ifdef ENABLE_EDITOR
135  Chronos::Engine::createTextureSampler(device, &textureSampler);
136  descriptorSet = ImGui_ImplVulkan_AddTexture(textureSampler, textureImageView,
137  VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL);
138 
139 #endif
140 }

181 void Chronos::Engine::transitionImageLayout(VkImage image,
182  VkImageLayout oldLayout, VkImageLayout newLayout,
183  VkCommandPool commandPool, Chronos::Engine::Device device)
184 {
185  VkCommandBuffer commandBuffer = Chronos::Engine::beginSingleTimeCommands(commandPool, device.device);
186  VkImageMemoryBarrier barrier {};
187  barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
188  barrier.oldLayout = oldLayout;
189  barrier.newLayout = newLayout;
190  barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
191  barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
192  barrier.image = image;
193  barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
194  barrier.subresourceRange.baseMipLevel = 0;
195  barrier.subresourceRange.levelCount = 1;
196  barrier.subresourceRange.baseArrayLayer = 0;
197  barrier.subresourceRange.layerCount = 1;
198  VkPipelineStageFlags sourceStage;
199  VkPipelineStageFlags destinationStage;
200 
201  if (oldLayout == VK_IMAGE_LAYOUT_UNDEFINED && newLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL) {
202  barrier.srcAccessMask = 0;
203  barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
204 
205  sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
206  destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
207  } else if (oldLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL && newLayout == VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL) {
208  barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
209  barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
210 
211  sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
212  destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
213  } else {
214  throw std::invalid_argument("unsupported layout transition!");
215  }
216 
217  vkCmdPipelineBarrier(commandBuffer, sourceStage, destinationStage, 0, 0,
218  nullptr, 0, nullptr, 1, &barrier);
219  Chronos::Engine::endSingleTimeCommands(&commandBuffer, device, commandPool);
220 }
