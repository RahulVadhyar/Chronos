66void Chronos::Engine::TexturedRectangle::createDescriptorSets()
67{
68 std::vector<VkDescriptorSetLayout> layouts(
69 MAX_FRAMES_IN_FLIGHT, descriptorSetLayout);
70 VkDescriptorSetAllocateInfo allocInfo {};
71 allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
72 allocInfo.descriptorPool = descriptorPool;
73 allocInfo.descriptorSetCount = static_cast<uint32_t>(MAX_FRAMES_IN_FLIGHT);
74 allocInfo.pSetLayouts = layouts.data();
75 descriptorSets.resize(MAX_FRAMES_IN_FLIGHT);
76 if (vkAllocateDescriptorSets(
77 device->device, &allocInfo, descriptorSets.data())
78 != VK_SUCCESS) {
79 throw std::runtime_error("failed to allocate descriptor sets!");
80 }
81 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
82 VkDescriptorBufferInfo bufferInfo {};
83 bufferInfo.buffer = uniformBuffers[i].buffer;
84 bufferInfo.offset = 0;
85 bufferInfo.range = sizeof(UniformBufferObject);
86
87 VkDescriptorImageInfo imageInfo {};
88 imageInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
89 imageInfo.imageView = texture.textureImageView;
90 imageInfo.sampler = textureSampler;
91
92 std::array<VkWriteDescriptorSet, 2> descriptorWrites {};
93 descriptorWrites[0].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
94 descriptorWrites[0].dstSet = descriptorSets[i];
95 descriptorWrites[0].dstBinding = 0;
96 descriptorWrites[0].dstArrayElement = 0;
97 descriptorWrites[0].descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
98 descriptorWrites[0].descriptorCount = 1;
99 descriptorWrites[0].pBufferInfo = &bufferInfo;
100
101 descriptorWrites[1].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
102 descriptorWrites[1].dstSet = descriptorSets[i];
103 descriptorWrites[1].dstBinding = 1;
104 descriptorWrites[1].dstArrayElement = 0;
105 descriptorWrites[1].descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
106 descriptorWrites[1].descriptorCount = 1;
107 descriptorWrites[1].pImageInfo = &imageInfo;
108
109 vkUpdateDescriptorSets(device->device,
110 static_cast<uint32_t>(descriptorWrites.size()),
111 descriptorWrites.data(), 0, nullptr);
112 }
113}

127Chronos::Engine::PipelineAttributes Chronos::Engine::TexturedRectangle::getPipelineAttributes()
128{
129 Chronos::Engine::PipelineAttributes pipelineAttributes;
130
131 auto bindingDescription = TexturedVertex::getBindingDescription();
132 auto attributeDescriptions = TexturedVertex::getAttributeDescriptions();
133
134 pipelineAttributes.bindingDescriptions.resize(1);
135 pipelineAttributes.bindingDescriptions[0] = bindingDescription;
136
137 pipelineAttributes.attributeDescriptions.resize(
138 attributeDescriptions.size());
139 for (int i = 0; i < static_cast<int>(attributeDescriptions.size()); i++) {
140 pipelineAttributes.attributeDescriptions[i] = attributeDescriptions[i];
141 }
142
143 pipelineAttributes.topology = VK_PRIMITIVE_TOPOLOGY_TRIANGLE_LIST;
144 pipelineAttributes.frontFace = VK_FRONT_FACE_COUNTER_CLOCKWISE;
145
146 pipelineAttributes.colorBlendAttachment.colorWriteMask
147 = VK_COLOR_COMPONENT_R_BIT | VK_COLOR_COMPONENT_G_BIT
148 | VK_COLOR_COMPONENT_B_BIT | VK_COLOR_COMPONENT_A_BIT;
149 pipelineAttributes.colorBlendAttachment.blendEnable = VK_FALSE;
150 pipelineAttributes.colorBlendAttachment.srcColorBlendFactor
151 = VK_BLEND_FACTOR_ONE;
152 pipelineAttributes.colorBlendAttachment.dstColorBlendFactor
153 = VK_BLEND_FACTOR_ZERO; // Optional
154 pipelineAttributes.colorBlendAttachment.colorBlendOp
155 = VK_BLEND_OP_ADD; // Optional
156 pipelineAttributes.colorBlendAttachment.srcAlphaBlendFactor
157 = VK_BLEND_FACTOR_ONE; // Optional
158 pipelineAttributes.colorBlendAttachment.dstAlphaBlendFactor
159 = VK_BLEND_FACTOR_ZERO; // Optional
160 pipelineAttributes.colorBlendAttachment.alphaBlendOp
161 = VK_BLEND_OP_ADD; // Optional
162
163 return pipelineAttributes;
164}

180 vkCmdBindDescriptorSets(commandBuffers[currentFrame],
181 VK_PIPELINE_BIND_POINT_GRAPHICS, this->pipelineLayout, 0, 1,
