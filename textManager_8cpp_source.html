52 void Chronos::Engine::TextManager::render(
53  uint32_t currentFrame, uint32_t imageIndex, float bgColor[3])
54 {
55  Chronos::Engine::ObjectManager<Chronos::Engine::Text>::render(
56  currentFrame, imageIndex, bgColor);
57 
58  VkViewport viewport {};
59  viewport.x = 0.0f;
60  viewport.y = 0.0f;
61  viewport.width = static_cast<float>(swapChain->swapChainExtent.width);
62  viewport.height = static_cast<float>(swapChain->swapChainExtent.height);
63  viewport.minDepth = 0.0f;
64  viewport.maxDepth = 1.0f;
65 
66  VkRect2D scissor {};
67  scissor.offset = { 0, 0 };
68  scissor.extent = swapChain->swapChainExtent;
69  for (auto& object : objects) {
70  if (this->objectsToBeRemoved.count(object.first) > 0)
71  continue;
72  vkCmdBindPipeline(commandBuffers[currentFrame],
73  VK_PIPELINE_BIND_POINT_GRAPHICS, object.second.graphicsPipeline);
74  vkCmdSetViewport(commandBuffers[currentFrame], 0, 1, &viewport);
75  vkCmdSetScissor(commandBuffers[currentFrame], 0, 1, &scissor);
76  VkBuffer vertexBuffers[] = { object.second.vertexBuffer };
77  VkDeviceSize offsets[] = { 0 };
78  vkCmdBindVertexBuffers(
79  commandBuffers[currentFrame], 0, 1, vertexBuffers, offsets);
80  vkCmdBindVertexBuffers(
81  commandBuffers[currentFrame], 1, 1, vertexBuffers, offsets);
82  vkCmdBindDescriptorSets(commandBuffers[currentFrame],
83  VK_PIPELINE_BIND_POINT_GRAPHICS, object.second.pipelineLayout, 0, 1,
84  &object.second.descriptorSets[currentFrame], 0, nullptr);
85  for (uint32_t j = 0; j < object.second.numLetters; j++) {
86  vkCmdDraw(commandBuffers[currentFrame], 4, 1, j * 4, 0);
87  }
88  }
89  Chronos::Engine::ObjectManager<Chronos::Engine::Text>::endRender(
90  currentFrame);
91 }
