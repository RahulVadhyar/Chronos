26 {
27  //this function is for textured triangles and rectangles only
28  //we need to add a check for the texture
29  std::vector<Chronos::Manager::TextureDetails> textureDetails = getTextureDetails();
30  bool textureExists = false;
31  for (auto& textureDetail : textureDetails) {
32  if (textureDetail.textureNo == texture) {
33  textureExists = true;
34  break;
35  }
36  }
37  if (!textureExists) {
38  throw std::runtime_error("Texture does not exist");
39  }
40 
41  //check whether shape name exists
42  for (auto& shape : engine.shapeManager.objects) {

44  throw std::runtime_error("Shape name already exists");
45  }
46  }
47  int shapeNo;
48  if (polygonType.triangle) {
49  shapeNo = engine.shapeManager.addTriangle(shapeParams, engine.textureManager.getTexture(texture));

52  } else {
53  throw std::runtime_error("Polygon type not supported");
54  }
55 
56  //we add 1 to shapeNo to know they are textured triangles or rectangles
